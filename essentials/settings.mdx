## Essential Properties

These are the fundamental settings required to get your documentation site up and running.

<ResponseField name="name" type="string" required>
The name of your project, used as the global title.

Example: `mintlify`
</ResponseField>

### Navigation Structure

<ResponseField name="navigation" type="Navigation[]" required>
  Define the structure of your documentation with groups and pages.
  <Expandable title="Navigation">
    <ResponseField name="group" type="string">
    Name of the navigation group.

    Example: `Settings`
    </ResponseField>
    <ResponseField name="pages" type="string[]">
    Relative paths to markdown files that will serve as pages.

    Example: `["customization", "page"]`
    </ResponseField>
  </Expandable>
</ResponseField>

<Tip>
  Organize your navigation groups logically based on topic areas to help users find content easily.
</Tip>

## Appearance Settings

### Logo Configuration

<ResponseField name="logo" type="string or object">
  Configure your site's logo for both light and dark modes.
  <Expandable title="Logo">
    <ResponseField name="light" type="string">
      Path to the logo used in light mode
    </ResponseField>
    <ResponseField name="dark" type="string">
      Path to the logo used in dark mode
    </ResponseField>
    <ResponseField name="href" type="string" default="/">
      Destination URL when clicking the logo
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="favicon" type="string">
  Path to your site's favicon image
</ResponseField>

### Color Theming

<ResponseField name="colors" type="Colors">
  Define your documentation's color scheme
  <Expandable title="Colors">
    <ResponseField name="primary" type="string" required>
      Primary color for highlights and accents in light mode
    </ResponseField>
    <ResponseField name="light" type="string">
      Primary color for dark mode elements
    </ResponseField>
    <ResponseField name="dark" type="string">
      Color for important buttons
    </ResponseField>
    <ResponseField name="background" type="object">
      Background colors for light and dark modes
      <Expandable title="Object">
        <ResponseField name="light" type="string" required>
          Background color in light mode
        </ResponseField>
        <ResponseField name="dark" type="string" required>
          Background color in dark mode
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

## Navigation Settings

### Top Navigation

<ResponseField name="topbarLinks" type="TopbarLink[]">
  Add links to the top navigation bar
  <Expandable title="TopbarLink">
    <ResponseField name="name" type="string">
    Button text

    Example: `Contact us`
    </ResponseField>
    <ResponseField name="url" type="string">
    Target URL for the button
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarCtaButton" type="Call to Action">
  Configure the main call-to-action button
  <Expandable title="Topbar Call to Action">
    <ResponseField name="type" type={'"link" or "github"'} default="link">
    Choose between a regular link or GitHub stars display
    </ResponseField>
    <ResponseField name="url" type="string">
    Button destination URL or GitHub repository URL
    </ResponseField>
    <ResponseField name="name" type="string">
    Button text (required for link type)
    </ResponseField>
  </Expandable>
</ResponseField>

### Version Control

<ResponseField name="versions" type="string[]">
  Enable version switching with an array of version names
</ResponseField>

<Note>
  Only configure versions if you need to maintain multiple documentation versions simultaneously.
</Note>

### Anchors and Tabs

<ResponseField name="anchors" type="Anchor[]">
  Configure navigation anchors with icons and grouping
  <Expandable title="Anchor">
    <ResponseField name="icon" type="string">
    Font Awesome icon name
    </ResponseField>
    <ResponseField name="name" type="string">
    Anchor label text
    </ResponseField>
    <ResponseField name="url" type="string">
    Base URL for the anchor section
    </ResponseField>
    <ResponseField name="color" type="string">
    Icon background color or gradient
    </ResponseField>
    <ResponseField name="version" type="string">
    Optional version restriction
    </ResponseField>
    <ResponseField name="isDefaultHidden" type="boolean" default="false">
    Hide anchor by default
    </ResponseField>
    <ResponseField name="iconType" default="duotone" type="string">
    Icon style variant
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topAnchor" type="Object">
  Customize the main documentation anchor
  <Expandable title="Object">
    <ResponseField name="name" default="Documentation" type="string">
      Main anchor text
    </ResponseField>
    <ResponseField name="icon" default="book-open" type="string">
      Main anchor icon
    </ResponseField>
    <ResponseField name="iconType" default="duotone" type="string">
      Icon style variant
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="tabs" type="Tabs[]">
  Create tabbed navigation sections
  <Expandable title="Tabs">
    <ResponseField name="name" type="string">
      Tab label text
    </ResponseField>
    <ResponseField name="url" type="string">
      Base URL for tab content
    </ResponseField>
  </Expandable>
</ResponseField>

## API Configuration

<ResponseField name="api" type="API">
  Settings for API documentation features
  <Expandable title="API">
    <ResponseField name="baseUrl" type="string">
      Base URL for API endpoints (can be array for multiple options)
    </ResponseField>

    <ResponseField name="auth" type="Auth">
      <Expandable title="Auth">
        <ResponseField name="method" type='"bearer" | "basic" | "key"'>
          Authentication method
        </ResponseField>
        <ResponseField name="name" type="string">
        Authentication parameter name
        </ResponseField>
        <ResponseField name="inputPrefix" type="string">
        Default authentication input prefix
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="playground" type="Playground">
      API playground settings
      <Expandable title="Playground">
        <ResponseField name="mode" default="show" type='"show" | "simple" | "hide"'>
          Playground display mode
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

### OpenAPI Integration

<ResponseField name="openapi" type="string | string[]">
  OpenAPI specification file location(s)
  
  <CodeGroup>
    ```json Absolute
    "openapi": "https://example.com/openapi.json"
    ```
    ```json Relative
    "openapi": "/openapi.json"
    ```
    ```json Multiple
    "openapi": ["https://example.com/openapi1.json", "/openapi2.json", "/openapi3.json"]
    ```
  </CodeGroup>
</ResponseField>

## Additional Features

### Social Media Integration

<ResponseField name="footerSocials" type="FooterSocials">
  Add social media links to the footer
  
  Example: 
  ```json
  {
    "x": "https://x.com/mintlify",
    "website": "https://mintlify.com"
  }
  ```
  <Expandable title="FooterSocials">
    <ResponseField name="[key]" type="string">
    Supported platforms: `website`, `facebook`, `x`, `discord`, `slack`, `github`, `linkedin`, `instagram`, `hacker-news`
    </ResponseField>
    <ResponseField name="property" type="string">
    Social platform URL
    </ResponseField>
  </Expandable>
</ResponseField>

### User Feedback

<ResponseField name="feedback" type="Feedback">
  Enable user feedback features
  <Expandable title="Feedback">
    <ResponseField name="suggestEdit" type="boolean" default="false">
    Allow users to suggest documentation edits
    </ResponseField>
    <ResponseField name="raiseIssue" type="boolean" default="false">
    Allow users to report documentation issues
    </ResponseField>
  </Expandable>
</ResponseField>

### Display Settings

<ResponseField name="modeToggle" type="ModeToggle">
  Configure dark/light mode settings
  <Expandable title="ModeToggle">
    <ResponseField name="default" type={'"light" or "dark"'}>
      Default color mode for new users
    </ResponseField>
    <ResponseField name="isHidden" type="boolean" default="false">
      Hide the mode toggle switch
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="backgroundImage" type="string">
  Add a background image to all pages
</ResponseField>

<Tip>
  For examples of background images in action, check out [Infisical](https://infisical.com/docs) and [FRPC](https://frpc.io).
</Tip>
